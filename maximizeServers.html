function maximizeServers(serverCapacities) {
    const freq = new Map();
    for (const c of serverCapacities) {
        freq.set(c, (freq.get(c) || 0) + 1);
    }

    const vals = [...freq.keys()].sort((a, b) => a - b);
    const n = vals.length;
    let best = 0;

    const get = (v) => freq.get(v) || 0;

    for (let i = 0; i < n; i++) {
        const v = vals[i];
        // single value
        best = Math.max(best, get(v));

        // two consecutive
        if (i + 1 < n && vals[i+1] === v + 1) {
            best = Math.max(best, get(v) + get(v+1));
        }

        // three consecutive: v, v+1, v+2
        // need count(v+1) >= count(v) + count(v+2) for all to be valid
        // otherwise take min subset: b + min(a+c, b)
        if (i + 1 < n && vals[i+1] === v+1 && i + 2 < n && vals[i+2] === v+2) {
            const a = get(v), b = get(v+1), c = get(v+2);
            if (b >= a + c) {
                best = Math.max(best, a + b + c);
            } else {
                // take all of b, and as many a+c as b allows
                best = Math.max(best, b + Math.min(a + c, b));
            }
        }
    }

    return best;
}
